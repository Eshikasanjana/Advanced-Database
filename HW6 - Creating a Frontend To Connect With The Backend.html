Python Code (app.py)


import streamlit as st
import mysql.connector
import pandas as pd
import os
from dotenv import load_dotenv

load_dotenv()

# DB Config
config = {
    'host': os.getenv("DB_HOST"),
    'port': int(os.getenv("DB_PORT")),
    'user': os.getenv("DB_USER"),
    'password': os.getenv("DB_PASSWORD"),
    'database': os.getenv("DB_NAME")
}

st.set_page_config(page_title="Shopping Dashboard", page_icon="üõçÔ∏è")
st.title("üõçÔ∏è Customer Shopping Dashboard")

try:
    conn = mysql.connector.connect(**config)
    st.success("Connected to database")
    cursor = conn.cursor()

    cursor.execute("SHOW TABLES;")
    tables = [t[0] for t in cursor.fetchall()]

    selected_table = st.selectbox("Choose a table", tables)

    df = pd.read_sql(f"SELECT * FROM {selected_table} LIMIT 10", conn)
    st.subheader(f"Preview of `{selected_table}` table")
    st.dataframe(df)

except Exception as e:
    st.error(f" Error: {e}")
finally:
    if 'cursor' in locals(): cursor.close()
    if 'conn' in locals(): conn.close()
